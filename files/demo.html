✕ toastr v2 (2015)

  • ✕ Requires jQuery (~87 KB overhead)
  • ✕ JS-driven animations (layout thrash)
  • ✕ No dark mode
  • ✕ No TypeScript types
  • ✕ Inaccessible (screen readers blind)
  • ✕ No ESM / tree-shaking
  • ✕ Gulp + LESS (abandoned toolchain)
  • ✕ No Promise API
  • ✕ No keyboard support
  • ✕ PNG sprite icons (extra HTTP request)

✓ toastr-next v3 (2026)

  • ✓ Zero dependencies — pure vanilla JS
  • ✓ CSS @keyframes — buttery smooth
  • ✓ Auto dark mode via prefers-color-scheme
  • ✓ Full TypeScript types + JSDoc
  • ✓ ARIA live regions, roles, keyboard nav
  • ✓ ESM + CJS + IIFE bundles
  • ✓ Vite build + Vitest tests
  • ✓ Promise API (await toast.onDismissed)
  • ✓ Escape key + focus management
  • ✓ CSS pseudo-element icons (no images)

// Install
npm install toastr-next

// Import (ESM)
import toastr from 'toastr-next';
import 'toastr-next/dist/toastr.css';

// Same API as before — no jQuery!
toastr.success('Saved!');
toastr.error('Failed', 'Oops');
toastr.info('New update');
toastr.warning('Low disk space', 'Warning');

// Global options
toastr.options = {
  positionClass: 'toast-bottom-right',
  animation: 'bounce',
  progressBar: true,
  closeButton: true,
};
// Returns a response object with a Promise
const toast = toastr.success('Changes saved!');

// Resolves when dismissed
await toast.onDismissed;
window.location.assign('/dashboard');

// Chain multiple toasts sequentially
async function multiStep() {
  await toastr.info('Step 1: Validating...').onDismissed;
  await toastr.info('Step 2: Uploading...').onDismissed;
  toastr.success('All done!');
}
// OLD (toastr v2 — jQuery required)
// <script src="jquery.min.js"></script>  ← 30 KB
// <script src="toastr.min.js"></script>  ← 5 KB
// Total: ~87 KB just to show a toast

// NEW (toastr-next v3 — zero deps)
// <link rel="stylesheet" href="toastr.css">
// <script src="toastr.js"></script>
// Total: ~6 KB including CSS

// API is identical — just remove jQuery
toastr.success('Hello!');  // works the same

// New options
toastr.options.animation = 'bounce';   // NEW
await toastr.info('Hi').onDismissed; // NEW
